Age Calculator

Requirements:

  • Ask user for name and birth year
  • Calculate age using current year
  • If age ≥ 18 → display "You are an Adult" else "You are a Minor"
  • Handle invalid inputs (empty, non-numeric, future year)

JavaScript Code:

function ageCalculator() {
    // Ask user for name
    let name = prompt("Enter your name:");
    
    // Check if name is empty or cancelled
    if (!name || name.trim() === "") {
        alert("Name is required!");
        return;
    }
    
    // Ask user for birth year
    let birthYear = prompt("Enter your birth year:");
    
    // Check if birth year is empty or cancelled
    if (!birthYear || birthYear.trim() === "") {
        alert("Birth year is required!");
        return;
    }
    
    // Convert birth year to number
    birthYear = parseInt(birthYear);
    
    // Check if birth year is a valid number
    if (isNaN(birthYear)) {
        alert("Please enter a valid numeric birth year!");
        return;
    }
    
    // Get current year
    let currentYear = new Date().getFullYear();
    
    // Check if birth year is in the future
    if (birthYear > currentYear) {
        alert("Birth year cannot be in the future!");
        return;
    }
    
    // Check if birth year is reasonable (not too old)
    if (birthYear < 1900) {
        alert("Please enter a reasonable birth year!");
        return;
    }
    
    // Calculate age
    let age = currentYear - birthYear;
    
    // Determine if adult or minor
    let status;
    if (age >= 18) {
        status = "You are an Adult";
    } else {
        status = "You are a Minor";
    }
    
    // Display result
    alert(`Hello ${name}!\nYour age is: ${age} years\n${status}`);
}
