BMI Categorization

Requirements:

  • Calculate user's body mass index (BMI)
  • Categorize BMI according to CDC guidelines
  • Handle invalid inputs

BMI Categories (CDC Guidelines):

BMI Weight Status
Below 18.5 Underweight
18.5 – 24.9 Normal
25.0 – 29.9 Overweight
30.0 and above Obese

JavaScript Code:

function bmiCalculator() {
    // Ask user for weight in kg
    let weight = prompt("Enter your weight in kilograms (kg):");
    
    // Check if weight is cancelled or empty
    if (!weight || weight.trim() === "") {
        alert("Weight is required!");
        return;
    }
    
    // Convert weight to number
    weight = parseFloat(weight);
    
    // Check if weight is a valid number
    if (isNaN(weight) || weight <= 0) {
        alert("Please enter a valid positive weight!");
        return;
    }
    
    // Ask user for height in meters
    let height = prompt("Enter your height in meters (m):");
    
    // Check if height is cancelled or empty
    if (!height || height.trim() === "") {
        alert("Height is required!");
        return;
    }
    
    // Convert height to number
    height = parseFloat(height);
    
    // Check if height is a valid number
    if (isNaN(height) || height <= 0) {
        alert("Please enter a valid positive height!");
        return;
    }
    
    // Calculate BMI
    let bmi = weight / (height * height);
    
    // Categorize BMI according to CDC guidelines
    let category;
    if (bmi < 18.5) {
        category = "Underweight";
    } else if (bmi >= 18.5 && bmi <= 24.9) {
        category = "Normal";
    } else if (bmi >= 25.0 && bmi <= 29.9) {
        category = "Overweight";
    } else {
        category = "Obese";
    }
    
    // Display result
    alert(`BMI Calculation Result:\nWeight: ${weight} kg\nHeight: ${height} m\nBMI: ${bmi.toFixed(2)}\nCategory: ${category}`);
}
