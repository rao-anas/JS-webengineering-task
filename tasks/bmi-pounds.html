BMI Calculation (Imperial Units)

Requirements:

  • Prompt user to enter weight in pounds and height in inches
  • Calculate BMI using imperial formula
  • Round result to tenths (1 decimal place)
  • Handle invalid inputs

BMI Formula (Imperial):

BMI = (weight in pounds × 703) ÷ (height in inches)²

JavaScript Code:

function bmiCalculatorPounds() {
    // Ask user for weight in pounds
    let weight = prompt("Enter your weight in pounds (lb):");
    
    // Check if weight is cancelled or empty
    if (!weight || weight.trim() === "") {
        alert("Weight is required!");
        return;
    }
    
    // Convert weight to number
    weight = parseFloat(weight);
    
    // Check if weight is a valid number
    if (isNaN(weight) || weight <= 0) {
        alert("Please enter a valid positive weight!");
        return;
    }
    
    // Ask user for height in inches
    let height = prompt("Enter your height in inches (in):");
    
    // Check if height is cancelled or empty
    if (!height || height.trim() === "") {
        alert("Height is required!");
        return;
    }
    
    // Convert height to number
    height = parseFloat(height);
    
    // Check if height is a valid number
    if (isNaN(height) || height <= 0) {
        alert("Please enter a valid positive height!");
        return;
    }
    
    // Calculate BMI using imperial formula: (weight in pounds × 703) ÷ (height in inches)²
    let bmi = (weight * 703) / (height * height);
    
    // Convert height to feet and inches for display
    let feet = Math.floor(height / 12);
    let inches = height % 12;
    
    // Display result rounded to tenths (1 decimal place)
    alert(`BMI Calculation Result (Imperial Units):\nWeight: ${weight} lbs\nHeight: ${height} inches (${feet}'${inches.toFixed(0)}")\nBMI: ${bmi.toFixed(1)}\n\nFormula: (weight in lbs × 703) ÷ (height in inches)²`);
}
