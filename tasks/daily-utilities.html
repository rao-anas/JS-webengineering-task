Daily Utilities Toolkit

Menu Options:

  • 1. Format Name (capitalize first letter of each word)
  • 2. Generate Random 4-digit OTP
  • 3. Search City in array ["Karachi", "Lahore", "Islamabad"]
  • 4. Show Today's Date in dd/mm/yyyy format
  • 5. Mini Calculator (add, subtract, multiply, divide)
  • 6. Exit

JavaScript Code:

function dailyUtilitiesToolkit() {
    let cities = ["Karachi", "Lahore", "Islamabad"];
    
    while (true) {
        let choice = prompt(`Daily Utilities Toolkit\n\nChoose an option:\n1. Format Name\n2. Generate 4-digit OTP\n3. Search City\n4. Show Today's Date\n5. Mini Calculator\n6. Exit\n\nEnter your choice (1-6):`);
        
        if (choice === null || choice === "6") {
            alert("Thank you for using Daily Utilities Toolkit!");
            break;
        }
        
        switch (choice) {
            case "1":
                formatName();
                break;
            case "2":
                generateOTP();
                break;
            case "3":
                searchCity(cities);
                break;
            case "4":
                showTodaysDate();
                break;
            case "5":
                miniCalculator();
                break;
            default:
                alert("Invalid option! Please enter a number between 1-6.");
        }
    }
}

function formatName() {
    let name = prompt("Enter a name to format:");
    if (!name || name.trim() === "") {
        alert("Name is required!");
        return;
    }
    
    // Split name into words and capitalize first letter of each word
    let formattedName = name.toLowerCase().split(' ').map(word => 
        word.charAt(0).toUpperCase() + word.slice(1)
    ).join(' ');
    
    alert(`Original: ${name}\nFormatted: ${formattedName}`);
}

function generateOTP() {
    // Generate random 4-digit OTP
    let otp = Math.floor(1000 + Math.random() * 9000);
    alert(`Your 4-digit OTP is: ${otp}`);
}

function searchCity(cities) {
    let searchCity = prompt("Enter city name to search:");
    if (!searchCity || searchCity.trim() === "") {
        alert("City name is required!");
        return;
    }
    
    // Search for city (case-insensitive)
    let found = false;
    for (let i = 0; i < cities.length; i++) {
        if (cities[i].toLowerCase() === searchCity.toLowerCase()) {
            found = true;
            break;
        }
    }
    
    if (found) {
        alert(`✓ City "${searchCity}" found in the list!`);
    } else {
        alert(`✗ City "${searchCity}" not found in the list.\nAvailable cities: ${cities.join(", ")}`);
    }
}

function showTodaysDate() {
    let today = new Date();
    let day = String(today.getDate()).padStart(2, '0');
    let month = String(today.getMonth() + 1).padStart(2, '0');
    let year = today.getFullYear();
    
    let formattedDate = `${day}/${month}/${year}`;
    alert(`Today's Date: ${formattedDate}`);
}

function miniCalculator() {
    let num1 = prompt("Enter first number:");
    if (!num1 || num1.trim() === "") {
        alert("First number is required!");
        return;
    }
    
    num1 = parseFloat(num1);
    if (isNaN(num1)) {
        alert("Please enter a valid first number!");
        return;
    }
    
    let operation = prompt("Choose operation:\n+ (Add)\n- (Subtract)\n* (Multiply)\n/ (Divide)\n\nEnter operation:");
    if (!operation || !["+", "-", "*", "/"].includes(operation)) {
        alert("Invalid operation! Please enter +, -, *, or /");
        return;
    }
    
    let num2 = prompt("Enter second number:");
    if (!num2 || num2.trim() === "") {
        alert("Second number is required!");
        return;
    }
    
    num2 = parseFloat(num2);
    if (isNaN(num2)) {
        alert("Please enter a valid second number!");
        return;
    }
    
    let result;
    switch (operation) {
        case "+":
            result = num1 + num2;
            break;
        case "-":
            result = num1 - num2;
            break;
        case "*":
            result = num1 * num2;
            break;
        case "/":
            if (num2 === 0) {
                alert("Error: Division by zero is not allowed!");
                return;
            }
            result = num1 / num2;
            break;
    }
    
    alert(`Calculation Result:\n${num1} ${operation} ${num2} = ${result}`);
}
