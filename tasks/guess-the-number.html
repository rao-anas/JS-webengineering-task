Guess the Number Game

Requirements:

  • Generate a random number (1‚Äì10)
  • Give user 3 attempts to guess
  • Show hints "Too High" or "Too Low"
  • End game if attempts run out or user cancels
  • Use random numbers, loops, conditionals, break, and flags

JavaScript Code:

function guessTheNumber() {
    // Generate random number between 1 and 10
    let randomNumber = Math.floor(Math.random() * 10) + 1;
    let attempts = 0;
    let maxAttempts = 3;
    let gameWon = false;
    
    alert("Welcome to Guess the Number Game!\nI have picked a number between 1 and 10.\nYou have 3 attempts to guess it!");
    
    // Game loop
    while (attempts < maxAttempts && !gameWon) {
        let guess = prompt(`Attempt ${attempts + 1}/${maxAttempts}\nEnter your guess (1-10):`);
        
        // Check if user cancelled
        if (guess === null) {
            alert("Game cancelled!");
            return;
        }
        
        // Check if input is empty
        if (guess.trim() === "") {
            alert("Please enter a number!");
            continue;
        }
        
        // Convert to number
        guess = parseInt(guess);
        
        // Check if input is a valid number
        if (isNaN(guess) || guess < 1 || guess > 10) {
            alert("Please enter a valid number between 1 and 10!");
            continue;
        }
        
        attempts++;
        
        // Check if guess is correct
        if (guess === randomNumber) {
            alert(`üéâ Congratulations! You guessed the number ${randomNumber} in ${attempts} attempt(s)!`);
            gameWon = true;
            break;
        } else if (attempts >= maxAttempts) {
            alert(`üòû Game Over! The number was ${randomNumber}. You used all ${maxAttempts} attempts.`);
        } else {
            // Give hint
            let hint = guess > randomNumber ? "Too High" : "Too Low";
            alert(`${hint}! You have ${maxAttempts - attempts} attempt(s) left.`);
        }
    }
}
