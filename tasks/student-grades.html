Student Marks & Grade System

Requirements:

  • Create an array of 5 subjects
  • Ask user to enter marks for each subject
  • Calculate total, percentage, and assign grade
  • Handle invalid marks (negative, >100, non-numeric)

Grading Scale:

Percentage Grade
≥ 80 A
≥ 60 B
< 60 Fail

JavaScript Code:

function studentGradeSystem() {
    // Create array of 5 subjects
    let subjects = ["Mathematics", "English", "Science", "History", "Computer Science"];
    let marks = [];
    let total = 0;
    
    alert("Student Grade System\nEnter marks for 5 subjects (out of 100)");
    
    // Ask user to enter marks for each subject
    for (let i = 0; i < subjects.length; i++) {
        let mark;
        let validInput = false;
        
        while (!validInput) {
            mark = prompt(`Enter marks for ${subjects[i]} (0-100):`);
            
            // Check if user cancelled
            if (mark === null) {
                alert("Grade calculation cancelled!");
                return;
            }
            
            // Check if input is empty
            if (mark.trim() === "") {
                alert("Marks are required!");
                continue;
            }
            
            // Convert to number
            mark = parseFloat(mark);
            
            // Check if input is a valid number
            if (isNaN(mark)) {
                alert("Please enter a valid numeric value!");
                continue;
            }
            
            // Check if marks are in valid range
            if (mark < 0 || mark > 100) {
                alert("Marks must be between 0 and 100!");
                continue;
            }
            
            validInput = true;
        }
        
        marks[i] = mark;
        total += mark;
    }
    
    // Calculate percentage
    let percentage = total / subjects.length;
    
    // Assign grade
    let grade;
    if (percentage >= 80) {
        grade = "A";
    } else if (percentage >= 60) {
        grade = "B";
    } else {
        grade = "Fail";
    }
    
    // Display formatted result
    let result = "Student Grade Report:\n\n";
    for (let i = 0; i < subjects.length; i++) {
        result += `${subjects[i]}: ${marks[i]}\n`;
    }
    result += `\nTotal Marks: ${total}/500\n`;
    result += `Percentage: ${percentage.toFixed(2)}%\n`;
    result += `Grade: ${grade}`;
    
    alert(result);
}
