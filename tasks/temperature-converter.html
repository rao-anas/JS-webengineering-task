Temperature Converter

Requirements:

  • Take user choice as input in a prompt box
  • Input a temperature, convert, and display with 2 decimal places
  • Handle invalid numeric inputs (NaN, empty, cancel)
  • User Choice 1: Fahrenheit to Celsius: (°F − 32) ÷ 1.8 = °C
  • User Choice 2: Celsius to Fahrenheit: (°C × 1.8) + 32 = °F

JavaScript Code:

function temperatureConverter() {
    // Ask user for conversion choice
    let choice = prompt("Choose conversion type:\n1. Fahrenheit to Celsius\n2. Celsius to Fahrenheit\n\nEnter 1 or 2:");
    
    // Check if choice is cancelled or empty
    if (!choice || choice.trim() === "") {
        alert("Conversion choice is required!");
        return;
    }
    
    // Check if choice is valid (1 or 2)
    if (choice !== "1" && choice !== "2") {
        alert("Invalid choice! Please enter 1 or 2.");
        return;
    }
    
    // Ask user for temperature value
    let temperature = prompt("Enter the temperature value:");
    
    // Check if temperature is cancelled or empty
    if (!temperature || temperature.trim() === "") {
        alert("Temperature value is required!");
        return;
    }
    
    // Convert temperature to number
    temperature = parseFloat(temperature);
    
    // Check if temperature is a valid number
    if (isNaN(temperature)) {
        alert("Please enter a valid numeric temperature!");
        return;
    }
    
    let convertedTemp;
    let fromUnit, toUnit, formula;
    
    if (choice === "1") {
        // Fahrenheit to Celsius: (°F − 32) ÷ 1.8 = °C
        convertedTemp = (temperature - 32) / 1.8;
        fromUnit = "°F";
        toUnit = "°C";
        formula = "(°F − 32) ÷ 1.8 = °C";
    } else {
        // Celsius to Fahrenheit: (°C × 1.8) + 32 = °F
        convertedTemp = (temperature * 1.8) + 32;
        fromUnit = "°C";
        toUnit = "°F";
        formula = "(°C × 1.8) + 32 = °F";
    }
    
    // Display result with 2 decimal places
    alert(`Temperature Conversion Result:\nOriginal: ${temperature}${fromUnit}\nConverted: ${convertedTemp.toFixed(2)}${toUnit}\nFormula: ${formula}`);
}
